    }
  }

  }

    self.last_name = last.to_string();
  }

  // Name to Tuple

    return (self.first_name,self.last_name);
  }
}

  p.set_last_name("Pro");
  println!("Person: {}",p.full_name());
  println!("Person: {:?}",p.to_tuple());
}
